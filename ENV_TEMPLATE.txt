# ==============================================================================
# QUALITY CONTROL SETTINGS (NEW!)
# ==============================================================================

# OPTIMIZED: Separate thresholds for each quality aspect
# This gives you fine-grained control over what matters most

# Hindi Quality (Grammar + Poetry): How perfect should the language be?
# - 9-10 = Perfect grammar, poetic (may need retries)
# - 8 = Excellent, minor issues okay (RECOMMENDED)
# - 7 = Good, some flexibility
HINDI_THRESHOLD=8

# Singability (Rhythm + Melody combined): How easy to sing?
# - 9-10 = Perfect meter, catchy hooks (may need retries)
# - 8 = Very singable, minor deviations okay
# - 7 = Singable, more flexibility (RECOMMENDED for cost savings)
# - 6 = Acceptable, prioritizes meaning over strict meter
SINGABILITY_THRESHOLD=7

# Cultural Sensitivity: How strict on religious accuracy?
# - 9-10 = Perfect adherence (very strict)
# - 8 = Respectful with minor creative liberty
# - 7 = Respectful, allows creativity (RECOMMENDED)
# - 6 = Basic respect, flexible
CULTURAL_THRESHOLD=7

# Max Quality Retries: How many times to regenerate if quality check fails
# - 3 = Maximum attempts (best quality, slowest)
# - 2 = Default (balanced)
# - 1 = One retry only (faster)
# - 0 = No retries (fastest, accept whatever is generated)
MAX_QUALITY_RETRIES=2

# Enable Quality Checks: Turn the entire quality system on/off
# - true = Check quality (default, recommended)
# - false = Skip all checks (ultra-fast mode, for testing only)
ENABLE_QUALITY_CHECKS=true

# Use Iterative Refinement: How to handle failed quality checks?
# - true = Refine existing lyrics (RECOMMENDED - saves cost, preserves good parts)
# - false = Regenerate from scratch (expensive, may lose good elements)
USE_ITERATIVE_REFINEMENT=true

# Skip Final Polish (Chain 4): Skip the final refinement pass?
# - true = Skip Chain 4 (faster, cheaper, rely on quality checks)
# - false = Run Chain 4 polish pass (slower, extra safety net)
SKIP_CHAIN4_REFINE=false

# Auto-Complete Truncated Lyrics: Retry when lyrics are incomplete?
# - true = Auto-detect and complete truncated lyrics (RECOMMENDED - better quality)
# - false = Accept partial lyrics without retry (faster, cheaper, useful for debugging)
# Cost: ~$0.0025 per retry (only when truncation detected, ~2% of songs)
AUTO_COMPLETE_TRUNCATED=true
